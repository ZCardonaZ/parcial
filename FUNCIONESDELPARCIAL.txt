1- CREATE OR REPLACE FUNCTION validar_correo(p_correo VARCHAR2)
RETURN VARCHAR2

IS

  v_valido VARCHAR2(3);
BEGIN
  
  IF LENGTH(p_correo) BETWEEN 10 AND 30 THEN

    IF REGEXP_LIKE(p_correo, '^[A-Za-z0-9]+@[A-Za-z0-9]+\.[a-z]+\.com\.co$') THEN
      v_valido := 'Sí';
    ELSE
      v_valido := 'No';
    END IF;
  ELSE
    v_valido := 'No';
  END IF;
  
  RETURN 'El correo ' || p_correo || ' es ' || v_valido || ' válido.';
END;
/


SELECT validar_correo('ejemplo@gmail.com') FROM dual;

create or replace NONEDITIONABLE FUNCTION es_multiplo_de_3(p_numero IN NUMBER)
RETURN VARCHAR2
IS
  v_resultado VARCHAR2(3);
BEGIN

  IF MOD(p_numero, 3) = 0 THEN
    v_resultado := 'Si es multi';
  ELSE
    v_resultado := 'No';
  END IF;

  RETURN 'El número ' || p_numero || ' es múltiplo de 3: ' || v_resultado;
END;
